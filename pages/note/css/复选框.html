  margin-right: 5px;
  cursor: pointer;
  font-size: 14px;

  position: relative;

  position: absolute;
  width: 10px;
  height: 15px;
  top: 0;
  content: " ";
  background-color: #ff0000;
  color: #fff;
  display: inline-block;
  visibility: visible;

}

input[type=checkbox]:checked:after {
  content: "✓";
  font-size: 12px;
}
